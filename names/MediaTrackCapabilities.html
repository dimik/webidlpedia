dictionary MediaTrackCapabilities {
  ULongRange width;

  DoubleRange aspectRatio;
  DoubleRange frameRate;
  sequence facingMode;
  sequence resizeMode;
  ULongRange sampleRate;

  sequence echoCancellation;
