HTML Standard defines HTMLElement

[Exposed=<a href='Window.html'>Window</a>]
interface <strong>HTMLElement</strong> : <a href='Element.html'>Element</a> {
  [HTMLConstructor] constructor();

  // metadata attributes
  [CEReactions] attribute <span>DOMString</span> <strong>title</strong>;
  [CEReactions] attribute <span>DOMString</span> <strong>lang</strong>;
  [CEReactions] attribute <span>boolean</span> <strong>translate</strong>;
  [CEReactions] attribute <span>DOMString</span> <strong>dir</strong>;

  // user interaction
  [CEReactions] attribute <span>boolean</span> <strong>hidden</strong>;
  <span>undefined</span> <strong>click</strong>();
  [CEReactions] attribute <span>DOMString</span> <strong>accessKey</strong>;
  readonly attribute <span>DOMString</span> <strong>accessKeyLabel</strong>;
  [CEReactions] attribute <span>boolean</span> <strong>draggable</strong>;
  [CEReactions] attribute <span>boolean</span> <strong>spellcheck</strong>;
  [CEReactions] attribute <span>DOMString</span> <strong>autocapitalize</strong>;

  [CEReactions] attribute [LegacyNullToEmptyString] <span>DOMString</span> <strong>innerText</strong>;

  <a href='ElementInternals.html'>ElementInternals</a> <strong>attachInternals</strong>();
};
